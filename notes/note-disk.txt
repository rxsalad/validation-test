
########## Check disks

df -h, check disk usage
du -sh *, check folder size

########## Check block devices

lsblk, list all block devices
lsblk -f

########## format

fdisk /dev/sda 
n - new partition
p - primary
1 - number 1
first to laster
w - write

mkfs.ext4 /dev/sda1 # format

########## mount

mkdir data 
mount /dev/sda1 /root/data

########## Auto-mount

blkid /dev/sda1 # get UUID

nano /etc/fstab
UUID=<ID> /root/data  ext4  defaults  0  2

mount -a # mount all in the /etc/fstab
